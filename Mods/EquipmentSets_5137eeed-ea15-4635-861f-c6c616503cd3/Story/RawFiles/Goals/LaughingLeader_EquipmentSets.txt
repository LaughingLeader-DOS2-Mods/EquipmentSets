INITSECTION
LLEQSET_Start_Internal_CompleteIfStarted(); // Existing saves
KBSECTION
PROC
LLEQSET_Internal_StartEquipmentSets()
THEN
DebugBreak("[LLEQSET] Equipment Sets starting.");
GoalCompleted;

IF
GameEventSet("GAMEEVENT_GameStarted")
THEN
LLEQSET_Internal_StartEquipmentSets();

PROC
LLEQSET_Start_Internal_CompleteIfStarted()
AND
DB_StoryStarted(_)
THEN
DebugBreak("[LLEQSET] Existing save detected.");
LLEQSET_Internal_StartEquipmentSets();

IF
DB_LeaderLib_GameStarted(_)
THEN
DebugBreak("[LLEQSET] Existing save detected. This goal wasn't completed after initializing, for some reason.");
LLEQSET_Internal_StartEquipmentSets();
EXITSECTION

ENDEXITSECTION
